.model small
.stack 100h
.data
    array db 10 dup(0)
.code
    main proc
        mov ax, @data
        mov ds, ax
        
        lea si, array
        mov [si], 0
        inc si
        mov [si], 1
        inc si
        
        mov cx, 08
        
generate:mov al, [si-1]
         add al, [si-2]
         mov [si], al
         
         inc si
         loop generate
         
         ;----displaying optional
         mov cx, 10 
         lea si, array
         mov ah, 02h
display: mov dl, [si]
         add dl, 48
         int 21h
         mov dl, ' '
         int 21h
         inc si
         loop display
         ;-----displaying ends
              
         
         mov ah, 4ch
         int 21h
              
    main endp              
 end main
