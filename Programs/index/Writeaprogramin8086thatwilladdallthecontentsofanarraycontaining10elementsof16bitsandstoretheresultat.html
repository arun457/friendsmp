.model small
.stack 100h
.data
    array dw 9999,9999,9999,9999,9999,9999,9999,9999,9999,9999
.code
    main proc
        mov ax, @data
        mov ds, ax ; initialize data segment
        
        lea si, array ; load effective address of array into SI
        mov cx, 10 ; set loop counter to 10
        mov bx, 0 ; set BX to 0 for summing
        mov dl, 0 ; for carry
        
    add_loop:
        add bx, [si] ; add contents of array element to BX
        jnc skip
        inc dl
   skip:add si, 2 ; increment SI to point to next element
        loop add_loop ; decrement CX and repeat until CX = 0
        
        mov [6000h], bx ; store result in memory location offset by 6000h
        mov [6002h], dl
        
        mov ah, 4Ch ; exit program
        int 21h
    main endp
end main
